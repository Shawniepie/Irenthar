		original_tag = LIR
		date > 1941.07.13
	}
	
	mean_time_to_happen = {
		days = 5
	}
	
	option = {#Monarchy!
		name = lir.5.a
		ai_chance = {
			factor = 5
			modifier = {
				monarchy_constitutional > 0.5
				factor = 2
			}
			modifier = {
				monarchy_constitutional > 0.6
				factor = 2
			}
			modifier = {
				monarchy_constitutional > 0.7
				factor = 2
			}
		}
		set_politics = {
			ruling_party = monarchy_constitutional
		}
		add_popularity = {
			popularity = 0.35
			ideology = monarchy_constitutional
		}
		set_country_flag = LIR_power_struggle_over
		if = {
			limit = {
				theocrat > 0.5
			}
			custom_effect_tooltip = LIR_civilwar_tt
			set_country_flag = LIR_theocrat_civilwar
		}
	}
	
	option = {#Theocracy!
		name = lir.5.b

				factor = 2
			}
			modifier = {
				theocrat > 0.7
				factor = 2
			}
		}
		set_politics = {
			ruling_party = theocrat
		}
		add_popularity = {
			popularity = 0.35
			ideology = theocrat
		}
